12 {

19  /* TODO: Add support for multiple RX TX queues using flow director

24 
25  rte_eth_dev_info_get(cfg->port_id, &dev_info);
26  RTE_LOG(INFO, PORT,
27  "[PORT INFO] Port ID: %d, Driver name: %s, Number of RX queues: "

32  // Use a generic port_conf by default

34  .rxmode = {
35  .max_rx_pkt_len = 9000,
36  },
37  .txmode = {
38  .mq_mode = ETH_MQ_TX_NONE,
39  },
40 
41  };
42  if (dev_info.tx_offload_capa & DEV_TX_OFFLOAD_MBUF_FAST_FREE) {

44  "[PORT INFO] Port ID: %d support fast mbuf free.\n",

48

50  cfg->tx_queues, &port_conf);
51 
52  if (ret < 0) {
53  rte_exit(EXIT_FAILURE,
54  "Cannot configure device: err=%d, port=%d\n", ret,
55  cfg->port_id);
56  }
57 
58  // Check that numbers of Rx and Tx descriptors satisfy descriptors
59  // limits from the ethernet device information, otherwise adjust them to
60  // boundaries.

63 
64  if (ret < 0) {
65  rte_exit(
66  EXIT_FAILURE,
67  "Cannot adjust number of descriptors: err=%d, port=%u\n",
68  ret, cfg->port_id);
69  }
70  rte_eth_macaddr_get(cfg->port_id, &port_eth_addr);
71 
72  // Allocate and setup RX and TX queues for a device
73  rte_eth_dev_info_get(cfg->port_id, &dev_info);
74 
75  rxq_conf = dev_info.default_rxconf;
76  rxq_conf.offloads = port_conf.rxmode.offloads;
77  ret = rte_eth_rx_queue_setup(cfg->port_id, 0, cfg->rx_descs,
78  rte_eth_dev_socket_id(cfg->port_id),
79  &rxq_conf, *(cfg->pool));
80  if (unlikely(ret != 0)) {
81  rte_exit(EXIT_FAILURE,
82  "Can not setup rx queue with error code:%d\n", ret);

84 
85  txq_conf = dev_info.default_txconf;

87  ret = rte_eth_tx_queue_setup(cfg->port_id, 0, cfg->tx_descs,
88  rte_eth_dev_socket_id(cfg->port_id),
89  &txq_conf);
90  if (unlikely(ret != 0)) {
91  rte_exit(EXIT_FAILURE,
92  "Can not setup tx queue with error code:%d\n", ret);
93  }
94 
95  // Start device
96  ret = rte_eth_dev_start(cfg->port_id);
97  if (ret < 0) {
98  rte_exit(EXIT_FAILURE, "Cannot start device: err=%d, port=%u\n",
99  ret, cfg->port_id);
100  }
101  rte_eth_promiscuous_enable(cfg->port_id);
102  RTE_LOG(INFO, PORT,
103  "Port:%d started, MAC address: %02X:%02X:%02X:%02X:%02X:%02X\n\n",
104  cfg->port_id, port_eth_addr.addr_bytes[0],
105  port_eth_addr.addr_bytes[1], port_eth_addr.addr_bytes[2],
106  port_eth_addr.addr_bytes[3], port_eth_addr.addr_bytes[4],
107  port_eth_addr.addr_bytes[5]);
108 
109  return 0;
110 }
